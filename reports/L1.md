## –ú–µ—Ç–æ–¥ –®–∏–º–±–µ–ª–ª–∞

–ü—É—Å—Ç—å –≥—Ä–∞—Ñ –∑–∞–¥–∞–Ω –º–∞—Ç—Ä–∏—Ü–µ–π $Œ© = (œâ_{ij})$ –≤–µ—Å–æ–≤ —Ä–µ–±–µ—Ä, —Ç.–µ.
$$ œâ_{ij} = \begin{cases}
	0, & \text{–µ—Å–ª–∏ –≤–µ—Ä—à–∏–Ω—ã }i, j\text{ –Ω–µ —Å–º–µ–∂–Ω—ã},\\
	œâ, & \text{–µ—Å–ª–∏ –≤–µ—Å —Ä–µ–±—Ä–∞ }ij = œâ.
\end{cases}
$$

–í–≤–µ–¥–µ–º —Å–ø–µ—Ü–∏–∞–ª—å–Ω—ã–µ –æ–ø–µ—Ä–∞—Ü–∏–∏ –Ω–∞–¥ —ç–ª–µ–º–µ–Ω—Ç–∞–º–∏ –º–∞—Ç—Ä–∏—Ü—ã
—Å–º–µ–∂–Ω–æ—Å—Ç–∏ –≤–µ—Ä—à–∏–Ω, –ø–æ–∑–≤–æ–ª—è—é—â–∏–µ –Ω–∞—Ö–æ–¥–∏—Ç—å –∫—Ä–∞—Ç—á–∞–π—à–∏–µ –∏–ª–∏ –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–µ –ø—É—Ç–∏ –º–µ–∂–¥—É
–≤–µ—Ä—à–∏–Ω–∞–º–∏,
—Å–æ—Å—Ç–æ—è—â–∏–µ –∏–∑ –∑–∞–¥–∞–Ω–Ω–æ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ —Ä–µ–±–µ—Ä. –≠—Ç–∏ –æ–ø–µ—Ä–∞—Ü–∏–∏ —Ç–∞–∫–æ–≤—ã.

1) –û–ø–µ—Ä–∞—Ü–∏—è —É–º–Ω–æ–∂–µ–Ω–∏—è –¥–≤—É—Ö –≤–µ–ª–∏—á–∏–Ω $a$ –∏ $b$ –ø—Ä–∏ –≤–æ–∑–≤–µ–¥–µ–Ω–∏–∏ –º–∞—Ç—Ä–∏—Ü—ã –≤ —Å—Ç–µ–ø–µ–Ω—å
—Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç –∏—Ö –∞–ª–≥–µ–±—Ä–∞–∏—á–µ—Å–∫–æ–π —Å—É–º–º–µ, —Ç–æ –µ—Å—Ç—å
	$$ \begin{cases}
   	a ‚ãÖ b ‚âù a+b\\
   	a ‚ãÖ 0 = 0 ‚ãÖ a ‚âù 0
 	\end{cases}
	$$

2) –û–ø–µ—Ä–∞—Ü–∏—è —Å–ª–æ–∂–µ–Ω–∏—è –¥–≤—É—Ö –≤–µ–ª–∏—á–∏–Ω $a$ –∏ $b$ –∑–∞–º–µ–Ω—è–µ—Ç—Å—è –≤—ã–±–æ—Ä–æ–º –∏–∑ —ç—Ç–∏—Ö –≤–µ–ª–∏—á–∏–Ω
–º–∏–Ω–∏–º–∞–ª—å–Ω–æ–≥–æ (–º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ) —ç–ª–µ–º–µ–Ω—Ç–∞, —Ç–æ –µ—Å—Ç—å
	\begin{equation}
	\begin{cases}
		a+b ‚âù \min (\max)\{a, b\}\\
		a + 0 = 0 + a ‚âù a
	\end{cases}\label{eq:shimbell_mul}
	\end{equation} 

	–Ω—É–ª–∏ –ø—Ä–∏ —ç—Ç–æ–º –∏–≥–Ω–æ—Ä–∏—Ä—É—é—Ç—Å—è. –ú–∏–Ω–∏–º–∞–ª—å–Ω—ã–π –∏–ª–∏ –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç –≤—ã–±–∏—Ä–∞–µ—Ç—Å—è –∏–∑
	–Ω–µ–Ω—É–ª–µ–≤—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤. –ù—É–ª—å –≤ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–µ –æ–ø–µ—Ä–∞—Ü–∏–∏ (\ref{eq:shimbell_mul}) –º–æ–∂–µ—Ç –±—ã—Ç—å –ø–æ–ª—É—á–µ–Ω –ª–∏—à—å —Ç–æ–≥–¥–∞,
	–∫–æ–≥–¥–∞ –≤—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã –∏–∑ –≤—ã–±–∏—Ä–∞–µ–º—ã—Ö --- –Ω—É–ª–µ–≤—ã–µ.

–° –ø–æ–º–æ—â—å—é —ç—Ç–∏—Ö –æ–ø–µ—Ä–∞—Ü–∏–π –¥–ª–∏–Ω—ã –∫—Ä–∞—Ç—á–∞–π—à–∏—Ö –∏–ª–∏ –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã—Ö –ø—É—Ç–µ–π –º–µ–∂–¥—É –≤—Å–µ–º–∏
–≤–µ—Ä—à–∏–Ω–∞–º–∏ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –≤–æ–∑–≤–µ–¥–µ–Ω–∏–µ–º –≤ —Å—Ç–µ–ø–µ–Ω—å –≤–µ—Å–æ–≤–æ–π –º–∞—Ç—Ä–∏—Ü—ã $Œ©$, —Å–æ–¥–µ—Ä–∂–∞—â–µ–π –≤–µ—Å–∞
—Ä–µ–±–µ—Ä. –ù–∞–ø—Ä–∏–º–µ—Ä, —ç–ª–µ–º–µ–Ω—Ç—ã –º–∞—Ç—Ä–∏—Ü—ã $Œ©^2 =\left(œâ^{(2)}_{ij}\right)$ –æ–ø—Ä–µ–¥–µ–ª—è—é—Ç—Å—è —Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º
$$ œâ^{(2)}_{ij} = \min_k (\max_k) \left\{œâ^{(1)}_{ik} + œâ^{(1)}_{kj}\right\}.
$$

–ê–Ω–∞–ª–æ–≥–∏—á–Ω–æ –æ–ø—Ä–µ–¥–µ–ª—è—é—Ç—Å—è —ç–ª–µ–º–µ–Ω—Ç—ã –º–∞—Ç—Ä–∏—Ü—ã $Œ©^m =\left(œâ^{(m)}_{ij}\right)$.
–î–ª–∏–Ω–∞ –∫—Ä–∞—Ç—á–∞–π—à–µ–≥–æ –∏–ª–∏ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –ø—É—Ç–∏ –∏–∑ $m$ —Ä–µ–±–µ—Ä –æ—Ç –≤–µ—Ä—à–∏–Ω—ã $i$ –¥–æ –≤–µ—Ä—à–∏–Ω—ã $j$ —Ä–∞–≤–Ω–∞ $œâ^{(m)}_{ij}$[@shaporev2006].

## –†–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –ü–æ–π–∞ 1

–†—è–¥ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –¥–∏—Å–∫—Ä–µ—Ç–Ω–æ–π —Å–ª—É—á–∞–π–Ω–æ–π –≤–µ–ª–∏—á–∏–Ω—ã $X$ --- —Å–æ–≤–æ–∫—É–ø–Ω–æ—Å—Ç—å
–≤—Å–µ—Ö –µ–µ –≤–æ–∑–º–æ–∂–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π $x_1,\dotsc,x_n$, –∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π $p_1,\dotsc,p_n$, –ø–æ—è–≤–ª–µ–Ω–∏—è –∫–∞–∂–¥–æ–≥–æ –∏–∑ —ç—Ç–∏—Ö –∑–Ω–∞—á–µ–Ω–∏–π.

–ü—Ä–∏–º–µ–º –∫–∞–∫ –æ–±–æ–∑–Ω–∞—á–µ–Ω–∏–µ $‚àè\limits_{k=0}^{-1} f(k) = 1$ (–∫–∞–∫ –ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏–µ –ø—É—Å—Ç–æ–≥–æ –º–Ω–æ–∂–µ—Å—Ç–≤–∞ —ç–ª–µ–º–µ–Ω—Ç–æ–≤).

–†—è–¥ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –ü–æ–π–∞ ---

$$ p(x) = C_n^x \frac {‚àè\limits_{k=0}^{x-1} (b+kc) ‚àè\limits_{k=0}^{n-x-1} (r+kc)} {‚àè\limits_{k=0}^{n-1}(b + r + kc)},
\ x=0,1,2,\dotsc, n.
$$	

–í —á–∞—Å—Ç–Ω–æ—Å—Ç–∏,
$$
p(0) = ‚àè_{k=0}^{n-1} \frac {r + kc}
         {b + r + kc},
\quad
p(n) = ‚àè_{k=0}^{n-1} \frac {b + kc}
         {b + r + kc},
$$

–≥–¥–µ $n>0$, $b>0$, $r>0$, $c ‚àà ‚Ñ§$. –ü–∞—Ä–∞–º–µ—Ç—Ä $c$ –º–æ–∂–µ—Ç –±—ã—Ç—å –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º,
–æ–¥–Ω–∞–∫–æ –æ–Ω –¥–æ–ª–∂–µ–Ω —É–¥–æ–≤–ª–µ—Ç–≤–æ—Ä—è—Ç—å —É—Å–ª–æ–≤–∏—é $b+r+c(n-1)>0$[@vadzinsky2001].

\begin{algorithm}
  \caption{–ì–µ–Ω–µ—Ä–∞—Ü–∏–∏ —Å–ª—É—á–∞–π–Ω–æ–≥–æ —á–∏—Å–ª–∞, –ø–æ–¥—á–∏–Ω—è—é—â–µ–≥–æ—Å—è —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—é –ü–æ–π–∞ 1}
  \label{alg:poja}
\begin{algorithmic}
\Require $n>0$, $b>0$, $r>0$, $c ‚àà ‚Ñ§$
\Ensure —Å–ª—É—á–∞–π–Ω–æ–µ —á–∏—Å–ª–æ –∏–∑ –º–Ω–æ–∂–µ—Å—Ç–≤–∞ $\{0,1,\dotsc, n\}$
\State $p := p_0$
\Comment{$p_0 = p(0)$}
\State $r :=$ \Call{Rand}{}
\Comment{$r ‚àà [0, 1]‚äÇ‚Ñù$, —Å–ª—É—á–∞–π–Ω–æ–µ —Å —Ä–∞–≤–Ω–æ–º–µ—Ä–Ω—ã–º —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ–º}
\For{$x := 0,n-1$}
	\State $r := r - p$
	\If{$r < 0$}
		\State \Return $x$
	\EndIf
	\State $p := p ‚ãÖ Œ±(x)$
	\Comment{$Œ±(x) = \dfrac {(n+1-x)\left(b + (x-1)c\right)}{x \left(r + (n-x)\right) c}$}
\EndFor
\State \Return $n$
\end{algorithmic}
\end{algorithm}

## –í—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω—ã–µ –ø—Ä–æ—Ü–µ–¥—É—Ä—ã

–í–æ –º–Ω–æ–≥–∏—Ö –∞–ª–≥–æ—Ä–∏—Ç–º–∞—Ö –ø–æ–∏—Å–∫–∞ –∫—Ä–∞—Ç—á–∞–π—à–∏—Ö –ø—É—Ç–µ–π –∏—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è –¥–≤–µ –ø—Ä–æ—Ü–µ–¥—É—Ä—ã.
–ü—Ä–æ—Ü–µ–¥—É—Ä–∞ –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏–∏ \textsc{Init}($s$) —Å—Ç—Ä–æ–∏—Ç –Ω–∞—á–∞–ª—å–Ω–æ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –º–∞—Ç—Ä–∏—Ü—ã –¥–ª–∏–Ω –ø—É—Ç–µ–π
–∏ –º–∞—Ç—Ä–∏—Ü—ã –ø—Ä–µ–¥—à–µ—Å—Ç–≤–æ–≤–∞–Ω–∏—è.

\begin{algorithmic}
\Require —É–∑–µ–ª $s$
\Ensure –∑–∞–ø–æ–ª–Ω–µ–Ω–Ω—ã–µ –º–∞—Ç—Ä–∏—Ü—ã $T$ :\Array{1..$p$, 1..$p$}{\Real} –¥–ª–∏–Ω –ø—É—Ç–µ–π –∏\\
 $Œ†$: \Array{1..$p$, 1..$p$}{0..$p$} —Å–∞–º–∏—Ö –ø—É—Ç–µ–π.
\Procedure{Init}{$s$}
\For{$v := 1,p$}
\State $T [v] := +‚àû$, $Œ†[v] := 0$
\EndFor
\State $T[s] := 0$
\EndProcedure
\end{algorithmic}

–ü—Ä–æ—Ü–µ–¥—É—Ä–∞ –æ—Å–ª–∞–±–ª–µ–Ω–∏—è, –∏–ª–∏ —Ä–µ–ª–∞–∫—Å–∞—Ü–∏–∏, ‚Äî \textsc{Relax}($s$, $v$, $u$) –ø—Ä–æ–≤–µ—Ä—è–µ—Ç, –≤–æ–∑–º–æ–∂–Ω–æ –ª–∏
—É–ª—É—á—à–∏—Ç—å –∏–∑–≤–µ—Å—Ç–Ω—ã–π –ø—É—Ç—å –∏–∑ —É–∑–ª–∞ $s$ –≤ —É–∑–µ–ª $v$, –ø—Ä–æ–≤–µ–¥—è –Ω–æ–≤—ã–π –ø—É—Ç—å —á–µ—Ä–µ–∑ —É–∑–µ–ª $u$,
–∏ –æ–±–Ω–æ–≤–ª—è–µ—Ç –ø—É—Ç–∏, –µ—Å–ª–∏ —ç—Ç–æ –≤–æ–∑–º–æ–∂–Ω–æ.

\begin{algorithmic}
\Require —É–∑–ª—ã $s$, $v$, $u$\\
–º–∞—Ç—Ä–∏—Ü–∞ $W$ : \Array{1..$p$, 1..$p$}{\Real} –¥–ª–∏–Ω –¥—É–≥,\\
–º–∞—Ç—Ä–∏—Ü–∞ $T$ : \Array{1..$p$, 1..$p$}{\Real} –¥–ª–∏–Ω –ø—É—Ç–µ–π,\\
–º–∞—Ç—Ä–∏—Ü–∞ $Œ†$ : \Array{1..$p$, 1..$p$}{0..$p$} —Å–∞–º–∏—Ö –ø—É—Ç–µ–π.
\Ensure –æ–±–Ω–æ–≤–ª–µ–Ω–Ω—ã–µ –º–∞—Ç—Ä–∏—Ü—ã $T$, $Œ†$
\Procedure{Relax}{$s$, $v$, $u$}
\If {$T [s, v] > T [s, u] + W [u, v]$}
\State $T [s, v] := T [s, u] + W [u, v]$; $Œ†[s, v] := u$ \Comment{–Ω–æ–≤—ã–π –ø—É—Ç—å –∫–æ—Ä–æ—á–µ}
\EndIf
\EndProcedure
\end{algorithmic}

–í–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ –ø—É—Ç–∏ –ø–æ –º–∞—Ç—Ä–∏—Ü–µ –ø—Ä–µ–¥—à–µ—Å—Ç–≤–æ–≤–∞–Ω–∏—è $Œ†$ –º–æ–∂–Ω–æ –ø—Ä–æ–∏–∑–≤–µ—Å—Ç–∏ —Ç–∞–∫:
<!-- 389 -->

\begin{algorithmic}
\State $w := v$; \Yield $w$ \Comment{–ø–æ—Å–ª–µ–¥–Ω–∏–π —É–∑–µ–ª}
\While{$w ‚â† u$}
	\State $w := Œ†[u, w]$
	\State \Yield $w$ \Comment{–ø—Ä–µ–¥—ã–¥—É—â–∏–π —É–∑–µ–ª}
\EndWhile
\end{algorithmic}

–ü–æ–ª—É—á–µ–Ω–Ω—É—é –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –º–æ–∂–Ω–æ –æ–±—Ä–∞—Ç–∏—Ç—å, –µ—Å–ª–∏ —ç—Ç–æ –ø–æ—Ç—Ä–µ–±—É–µ—Ç—Å—è, –∏ –ø–æ–ª—É—á–∏—Ç—å –ø—Ä—è–º–æ–π
–ø–æ—Ä—è–¥–æ–∫ —É–∑–ª–æ–≤ –ø—É—Ç–∏.

## –ê–ª–≥–æ—Ä–∏—Ç–º –î–µ–π–∫—Å—Ç—Ä—ã

–ê–ª–≥–æ—Ä–∏—Ç–º –î–µ–π–∫—Å—Ç—Ä—ã \ref{alg:dijkstra} –Ω–∞—Ö–æ–¥–∏—Ç –∫—Ä–∞—Ç—á–∞–π—à–∏–π –ø—É—Ç—å –º–µ–∂–¥—É –¥–≤—É–º—è –¥–∞–Ω–Ω—ã–º–∏ –≤–µ—Ä—à–∏–Ω–∞–º–∏
(—É–∑–ª–∞–º–∏) –≤ (–æ—Ä)–≥—Ä–∞—Ñ–µ, –µ—Å–ª–∏ –¥–ª–∏–Ω—ã –¥—É–≥ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã[@novikov2008].

_–°–ª–æ–∂–Ω–æ—Å—Ç—å –∞–ª–≥–æ—Ä–∏—Ç–º–∞_: $ùí™(p^2)$.

\begin{algorithm}
  \caption{–î–µ–π–∫—Å—Ç—Ä—ã –ø–æ–∏—Å–∫–∞ –∫—Ä–∞—Ç—á–∞–π—à–∏—Ö –ø—É—Ç–µ–π}\label{alg:dijkstra}
  \begin{algorithmic}
  \Require
  	–≤–∑–≤–µ—à–µ–Ω–Ω—ã–π –æ—Ä–≥—Ä–∞—Ñ $G(V, E)$,\\
  	–º–∞—Ç—Ä–∏—Ü–∞ –≤–µ—Å–æ–≤ $W$ : \Array{1..$p$, 1..$p$}{\Real},\\
		–∏—Å—Ç–æ—á–Ω–∏–∫ $s$.
  \Ensure
  	–≤–µ–∫—Ç–æ—Ä $T$ : \Array{1..$p$}{\Real} –¥–ª–∏–Ω –∫—Ä–∞—Ç—á–∞–π—à–∏—Ö –ø—É—Ç–µ–π –æ—Ç –∏—Å—Ç–æ—á–Ω–∏–∫–∞,\\
  	–≤–µ–∫—Ç–æ—Ä $Œ†$ : \Array{1..$p$}{0..$p$} —Å–∞–º–∏—Ö –ø—É—Ç–µ–π.
  \State \Call{Init}{$s$} \Comment{–∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è}
  \State $Q:=V$ \Comment{–∫–æ–Ω—Ç–µ–π–Ω–µ—Ä (–æ—á–µ—Ä–µ–¥—å —Å –ø—Ä–∏–æ—Ä–∏—Ç–µ—Ç–∞–º–∏)}
  \While{$Q‚â†‚àÖ$}
		\State $u$ := \Call{ExtractMin}{$Q$} \Comment{–∏–∑–≤–ª–µ—á–µ–Ω–∏–µ —É–∑–ª–∞ —Å –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–º –∑–Ω–∞—á–µ–Ω–∏–µ–º $T[u]$}
		\If{$T[u] = ‚àû$} \Stop \EndIf \Comment{–æ—Å—Ç–∞–ª—å–Ω—ã–µ —É–∑–ª—ã –Ω–µ–¥–æ—Å—Ç–∏–∂–∏–º—ã –∏–∑ $s$}
		\ForAll{$v ‚àà Œì(u)$}
			\If{$v ‚àà Q$}
				\State \Call{Relax}{$s$, $v$, $u$} \Comment{—Ä–µ–ª–∞–∫—Å–∞—Ü–∏—è –¥—É–≥–∏ $(u, v)$}
			\EndIf
		\EndFor
  \EndWhile
  \end{algorithmic}
\end{algorithm}

## –ê–ª–≥–æ—Ä–∏—Ç–º –ë–µ–ª–ª–º–∞–Ω–∞-–§–æ—Ä–¥–∞

–ê–ª–≥–æ—Ä–∏—Ç–º –ë–µ–ª–ª–º–∞–Ω–∞-–§–æ—Ä–¥–∞ \ref{alg:bellman-ford} –Ω–∞—Ö–æ–¥–∏—Ç –∫—Ä–∞—Ç—á–∞–π—à–∏–µ –ø—É—Ç–∏ –≤–æ –≤–∑–≤–µ—à–µ–Ω–Ω–æ–º –≥—Ä–∞—Ñ–µ,
–Ω–µ —Å–æ–¥–µ—Ä–∂–∞—â–µ–º —Ü–∏–∫–ª—ã —Å –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º —Å—É–º–º–∞—Ä–Ω—ã–º –≤–µ—Å–æ–º, –æ—Ç –æ–¥–Ω–æ–π –≤–µ—Ä—à–∏–Ω—ã –¥–æ –≤—Å–µ—Ö –æ—Å—Ç–∞–ª—å–Ω—ã—Ö[@shaporev2006].

## –ú–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π –ø–æ—Ç–æ–∫ –≤ —Å–µ—Ç–∏

–ü—É—Å—Ç—å $G(V,E,c)$ --- —Å–µ—Ç—å, $s$ –∏ $t$ --- c–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ, –∏—Å—Ç–æ—á–Ω–∏–∫ –∏ —Å—Ç–æ–∫ —Å–µ—Ç–∏.
–î—É–≥–∏ —Å–µ—Ç–∏ –Ω–∞–≥—Ä—É–∂–µ–Ω—ã –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º–∏ –≤–µ—â–µ—Å—Ç–≤–µ–Ω–Ω—ã–º–∏ —á–∏—Å–ª–∞–º–∏, $c:\ E ‚Üí ‚Ñù_+$.
–ï—Å–ª–∏ $u$ –∏ $v$ --- —É–∑–ª—ã —Å–µ—Ç–∏, —Ç–æ —á–∏—Å–ª–æ $c(u,v)$ --- –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è _–ø—Ä–æ–ø—É—Å–∫–Ω–æ–π —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç—å—é_ –¥—É–≥–∏ $(u,v)$. 

_–î–∏–≤–µ—Ä–≥–µ–Ω—Ü–∏–µ–π —Ñ—É–Ω–∫—Ü–∏–∏_ $f$ –≤ —É–∑–ª–µ $v$ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è —á–∏—Å–ª–æ $\div(f, v)$, –∫–æ—Ç–æ—Ä–æ–µ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è —Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º:

$$ \div(f,u) ‚âù ‚àë_{v|(u,v) ‚àà E} f(u,v) \quad- ‚àë_{v|(u,v) ‚àà E} f(v,u)
$$

–§—É–Ω–∫—Ü–∏—è $f: E ‚Üí ‚Ñù$ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è _–ø–æ—Ç–æ–∫–æ–º_ –≤ —Å–µ—Ç–∏ $G$, –µ—Å–ª–∏:

1. $‚àÄ (u,v) ‚àà E (0 ‚©Ω f(u,v) ‚©Ω c(u,v))$,
		—Ç–æ –µ—Å—Ç—å –ø–æ—Ç–æ–∫ —á–µ—Ä–µ–∑ –¥—É–≥—É –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–µ–Ω –∏ –Ω–µ –ø—Ä–µ–≤–æ—Å—Ö–æ–¥–∏—Ç –ø—Ä–æ–ø—É—Å–∫–Ω–æ–π —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç–∏ –¥—É–≥–∏.
2. $‚àÄ u ‚àà V ‚ßµ \{s,t\} (\div(f,u)=0)$, —Ç–æ –µ—Å—Ç—å –¥–∏–≤–µ—Ä–≥–µ–Ω—Ü–∏—è –ø–æ—Ç–æ–∫–∞ —Ä–∞–≤–Ω–∞ –Ω—É–ª—é –≤–æ –≤—Å–µ—Ö —É–∑–ª–∞—Ö,
		–∫—Ä–æ–º–µ –∏—Å—Ç–æ—á–Ω–∏–∫–∞ –∏ —Å—Ç–æ–∫–∞.

_–í–µ–ª–∏—á–∏–Ω–∞ –ø–æ—Ç–æ–∫–∞ –≤ —Å–µ—Ç–∏_ $G$ --- —Å—É–º–º–∞ –≤—Å–µ—Ö –ø–æ—Ç–æ–∫–æ–≤,
–≤—ã—Ö–æ–¥—è—â–∏—Ö –∏–∑ –∏—Å—Ç–æ–∫–∞, —Ç–æ –µ—Å—Ç—å[@novikov2008]
$$ w(f) = \div(f,s).
$$

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø–∞—Ä—É
–≤–µ—Ä—à–∏–Ω $u,v‚ààV$. –û–ø—Ä–µ–¥–µ–ª–∏–º _–æ—Å—Ç–∞—Ç–æ—á–Ω—É—é –ø—Ä–æ–ø—É—Å–∫–Ω—É—é —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç—å_
$c_f(u,v)$ –∫–∞–∫
$$ c_f(u,v) = \begin{cases}
	c(u,v) - f(u,v), & \text{–µ—Å–ª–∏ }(u,v)‚ààE,\\
	f(v,u),          & \text{–µ—Å–ª–∏ }(v,u)‚ààE,\\

\end{cases}
$$

## –ê–ª–≥–æ—Ä–∏—Ç–º –§–æ—Ä–¥–∞-–§–∞–ª–∫–µ—Ä—Å–æ–Ω–∞

_–¢–µ–æ—Ä–µ–º–∞ –§–æ—Ä–¥–∞-–§–∞–ª–∫–µ—Ä—Å–æ–Ω–∞_ –ú–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π –ø–æ—Ç–æ–∫ –≤ —Å–µ—Ç–∏ —Ä–∞–≤–µ–Ω –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–π –ø—Ä–æ–ø—É—Å–∫–Ω–æ–π —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç–∏ —Ä–∞–∑—Ä–µ–∑–∞, —Ç–æ –µ—Å—Ç—å —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –ø–æ—Ç–æ–∫ $f^*$ —Ç–∞–∫–æ–π, —á—Ç–æ 
$$ w(f^*) = \max_f w(f) = \min_P C(P).
$$

–ù–∞ –æ—Å–Ω–æ–≤–µ –¥–∞–Ω–Ω–æ–π —Ç–µ–æ—Ä–µ–º—ã —Ä–µ–∞–ª–∏–∑—É–µ—Ç—Å—è –∞–ª–≥–æ—Ä–∏—Ç–º –§–æ—Ä–¥–∞-–§–∞–ª–∫–µ—Ä—Å–æ–Ω–∞ –¥–ª—è –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –ø–æ—Ç–æ–∫–∞ –≤ —Å–µ—Ç–∏,
–∑–∞–¥–∞–Ω–Ω–æ–π –º–∞—Ç—Ä–∏—Ü–µ–π –ø—Ä–æ–ø—É—Å–∫–Ω—ã—Ö —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç–µ–π –¥—É–≥.

–ê–ª–≥–æ—Ä–∏—Ç–º –§–æ—Ä–¥–∞-–§–∞–ª–∫–µ—Ä—Å–æ–Ω–∞ —Ä–µ—à–∞–µ—Ç –∑–∞–¥–∞—á—É –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏—è –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –ø–æ—Ç–æ–∫–∞ –≤ —Ç—Ä–∞–Ω—Å–ø–æ—Ä—Ç–Ω–æ–π —Å–µ—Ç–∏[@novikov2008].

–ü—Ä–∏–≤–µ–¥–µ–Ω–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è \ref{alg:ffulk} –º–µ—Ç–æ–¥–∞ –§–æ—Ä–¥–∞-–§–∞–ª–∫–µ—Ä—Å–æ–Ω–∞ –≤—ã—á–∏—Å–ª—è–µ—Ç –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π –ø–æ—Ç–æ–∫ –≤ —Ç—Ä–∞–Ω—Å–ø–æ—Ä—Ç–Ω–æ–π —Å–µ—Ç–∏
$G(V, E)$ –ø—É—Ç–µ–º –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è
–∞—Ç—Ä–∏–±—É—Ç–∞ –ø–æ—Ç–æ–∫–∞ $(u,v).f$ –∫–∞–∂–¥–æ–≥–æ —Ä–µ–±—Ä–∞ $(u,v)‚ààE$.
–ï—Å–ª–∏ $(u,v)‚àâE$, –Ω–µ—è–≤–Ω–æ –ø—Ä–µ–¥–ø–æ–ª–∞–≥–∞–µ—Ç—Å—è, —á—Ç–æ $(u,v).f=0$.

\begin{algorithm}
\caption{–§–æ—Ä–¥–∞-–§–∞–ª–∫–µ—Ä—Å–æ–Ω–∞ –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏—è –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –ø–æ—Ç–æ–∫–∞ –≤ —Å–µ—Ç–∏}\label{alg:ffulk}
\begin{algorithmic}
\Require –ì—Ä–∞—Ñ $G(V,E)$,
–∏—Å—Ç–æ—á–Ω–∏–∫ $s$ –∏ —Å—Ç–æ–∫ $t$ —Å–µ—Ç–∏.

–ü—Ä–æ–ø—É—Å–∫–Ω—ã–µ —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç–∏ $c(u,v)$, $c(u,v) = 0$ –µ—Å–ª–∏ $(u,v)‚àâE$.
\Ensure 
\For{$(u,v)‚ààE$}
	\State $(u,v).f = 0$
\EndFor
\While{—Å—É—â–µ—Å—Ç–≤—É–µ—Ç –ø—É—Ç—å $p$ –∏–∑ $s$ –≤ $t$ –≤ –æ—Å—Ç–∞—Ç–æ—á–Ω–æ–π —Å–µ—Ç–∏ $G_f$}
	\State $c_f(p) := \min \{ c_f(u,v) : (u,v)‚ààp \}$ \Comment{$c_f(p)$ --- –ª–æ–∫–∞–ª—å–Ω–∞—è –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è —Ü–∏–∫–ª–∞}
	\For{$(u,v) ‚àà p$}
		\If{$(u,v)‚ààE$}
			 $(u,v).f := (u,v).f + c_f(p)$
		\Else
			 $\ (v,u).f := (v,u).f - c_f(p)$
		\EndIf
	\EndFor
\EndWhile
\end{algorithmic}
\end{algorithm}

–ú–æ–∂–Ω–æ —É–ª—É—á—à–∏—Ç—å –≤—Ä–µ–º–µ–Ω–Ω—É—é –≥—Ä–∞–Ω–∏—Ü—É –∞–ª–≥–æ—Ä–∏—Ç–º–∞ \ref{alg:ffulk}, –µ—Å–ª–∏ —Ä–µ–∞–ª–∏–∑–æ–≤–∞—Ç—å –≤—ã—á–∏—Å–ª–µ–Ω–∏–µ —É–≤–µ–ª–∏—á–∏–≤–∞—é—â–µ–≥–æ –ø—É—Ç–∏ $p$ –≤ —Å—Ç—Ä–æ–∫–µ __while__ –∫–∞–∫ –ø–æ–∏—Å–∫ –≤ —à–∏—Ä–∏–Ω—É,

–≤ –æ—Å—Ç–∞—Ç–æ—á–Ω–æ–π —Å–µ—Ç–∏, –≥–¥–µ –∫–∞–∂–¥–æ–µ —Ä–µ–±—Ä–æ –∏–º–µ–µ—Ç –µ–¥–∏–Ω–∏—á–Ω—É—é –¥–ª–∏–Ω—É (–≤–µ—Å). –¢–∞–∫–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è –º–µ—Ç–æ–¥–∞ –§–æ—Ä–¥–∞-–§–∞–ª–∫–µ—Ä—Å–æ–Ω–∞ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è _–∞–ª–≥–æ—Ä–∏—Ç–º–æ–º –≠–¥–º–æ–Ω–¥—Å–∞-–ö–∞—Ä–ø–∞_
(Edmonds-Karp algorithm)[@cormen2013].

_–°–ª–æ–∂–Ω–æ—Å—Ç—å –∞–ª–≥–æ—Ä–∏—Ç–º–∞_: $ùí™(p‚ãÖq^2)$.

## –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –ø–æ—Ç–æ–∫–∞ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–π —Å—Ç–æ–∏–º–æ—Å—Ç–∏

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º –∑–∞–¥–∞—á—É –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –ø–æ—Ç–æ–∫–∞ –∑–∞–¥–∞–Ω–Ω–æ–π –≤–µ–ª–∏—á–∏–Ω—ã $Œ∏$ –æ—Ç $s$ –∫ $t$ –≤ —Å–µ—Ç–∏
$G(V,E,c,d)$, –≤ –∫–æ—Ç–æ—Ä–æ–π –∫–∞–∂–¥–∞—è –¥—É–≥–∞ $(u,v) ‚àà E$ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É–µ—Ç—Å—è –Ω–µ —Ç–æ–ª—å–∫–æ –ø—Ä–æ–ø—É—Å–∫–Ω–æ–π —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç—å—é
$c(u,v)$, –Ω–æ –∏ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–æ–π —Å—Ç–æ–∏–º–æ—Å—Ç—å—é $d(u,v)$ –ø–µ—Ä–µ—Å—ã–ª–∫–∏ –µ–¥–∏–Ω–∏—á–Ω–æ–≥–æ –ø–æ—Ç–æ–∫–∞ –∏–∑
$i$ –≤ $j$ –≤–¥–æ–ª—å –¥—É–≥–∏ $(u,v)$. $d:\ E ‚Üí ‚Ñù_+$.

–ï—Å–ª–∏ $Œ∏>f_{\max}$, –≥–¥–µ $f_{\max}$ --- –≤–µ–ª–∏—á–∏–Ω–∞ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –ø–æ—Ç–æ–∫–∞ –≤ —Å–µ—Ç–∏ $G$ –æ—Ç $s$ –∫ $t$,
—Ç–æ —Ä–µ—à–µ–Ω–∏—è –Ω–µ—Ç.
–ï—Å–ª–∏ –∂–µ $Œ∏‚©Ωf_{\max}$, —Ç–æ –º–æ–∂–µ—Ç –±—ã—Ç—å –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–æ –Ω–µ—Å–∫–æ–ª—å–∫–æ —Ä–∞–∑–ª–∏—á–Ω—ã—Ö –ø–æ—Ç–æ–∫–æ–≤ –≤–µ–ª–∏—á–∏–Ω—ã
$Œ∏$ –æ—Ç $s$ –∫ $t$.

–ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∞—è –º–æ–¥–µ–ª—å –∑–∞–¥–∞—á–∏ ---
–º–∏–Ω–∏–º–∏–∑–∏—Ä–æ–≤–∞—Ç—å —Ü–µ–ª–µ–≤—É—é —Ñ—É–Ω–∫—Ü–∏—é –≤—ã–±–æ—Ä–æ–º –ø–æ—Ç–æ–∫–∞ $f$:

$$ ‚àë_{(u,v) ‚àà E} d(u,v) ‚ãÖ f(u,v) ‚Üí \min_{f},
$$

–≥–¥–µ $d(u,v)$ --- –≤–µ—Å –¥—É–≥–∏ $(u,v)$, $f(u,v)$ --- –≤–µ–ª–∏—á–∏–Ω–∞ –ø–æ—Ç–æ–∫–∞ –≤–¥–æ–ª—å –¥—É–≥–∏ $(u,v)$[@shaporev2006].

## –ó–∞–¥–∞—á–∞ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –ø–æ—Ç–æ–∫–∞ –≤ —Å–µ—Ç–∏ —Å –Ω–µ—Å–∫–æ–ª—å–∫–∏–º–∏ –∏—Å—Ç–æ–∫–∞–º–∏ –∏ –Ω–µ—Å–∫–æ–ª—å–∫–∏–º–∏ —Å—Ç–æ–∫–∞–º–∏

–í –∑–∞–¥–∞—á–µ –æ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–º –ø–æ—Ç–æ–∫–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –Ω–µ—Å–∫–æ–ª—å–∫–æ –∏—Å—Ç–æ–∫–æ–≤ –∏ —Å—Ç–æ–∫–æ–≤. –ù–∞–ø—Ä–∏–º–µ—Ä,
$m$ –∏—Å—Ç–æ–∫–æ–≤ $\{ s_1, s_2, \dotsc, s_m\}$ –∏
$n$ —Å—Ç–æ–∫–æ–≤ $\{t_1,t_2, \dotsc,t_n\}$.

–ó–∞–¥–∞—á–∞ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–º –ø–æ—Ç–æ–∫–∞ –≤ —Å–µ—Ç–∏ —Å –Ω–µ—Å–∫–æ–ª—å–∫–∏–º–∏ –∏—Å—Ç–æ–∫–∞–º–∏
–∏ –Ω–µ—Å–∫–æ–ª—å–∫–∏–º–∏ —Å—Ç–æ–∫–∞–º–∏ —Å–≤–æ–¥–∏—Ç—Å—è –∫ –æ–±—ã—á–Ω–æ–π –∑–∞–¥–∞—á–µ –æ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–º –ø–æ—Ç–æ–∫–µ.
–î–ª—è —ç—Ç–æ–≥–æ –≤ —Å–µ—Ç—å –¥–æ–±–∞–≤–ª—è–µ—Ç—Å—è _—Ñ–∏–∫—Ç–∏–≤–Ω—ã–π –∏—Å—Ç–æ–∫_ (supersource) $s$
–∏ –æ—Ä–∏–µ–Ω—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ —Ä–µ–±—Ä–∞ $(s, s_i)$ —Å –ø—Ä–æ–ø—É—Å–∫–Ω–æ–π —Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç—å—é
$$ c(s,s_i)=‚àû, \qquad i = 1,\dotsc,m.
$$
–¢–æ—á–Ω–æ —Ç–∞–∫ –∂–µ —Å–æ–∑–¥–∞–µ—Ç—Å—è _—Ñ–∏–∫—Ç–∏–≤–Ω—ã–π —Å—Ç–æ–∫_ (supresink) $t$ –∏ –¥–æ–±–∞–≤–ª—è—é—Ç—Å—è –æ—Ä–∏–µ–Ω—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ —Ä–µ–±—Ä–∞ $(t_i, t)$, –æ–±–ª–∞–¥–∞—é—â–∏–µ
$$ c(t_i , t) = ‚àû, \qquad i=1,\dotsc,n.
$$

–ï–¥–∏–Ω—Å—Ç–≤–µ–Ω–Ω—ã–π –∏—Å—Ç–æ–∫ $s$ –æ–±–µ—Å–ø–µ—á–∏–≤–∞–µ—Ç –ø–æ—Ç–æ–∫ –ª—é–±–æ–π —Ç—Ä–µ–±—É–µ–º–æ–π –≤–µ–ª–∏—á–∏–Ω—ã –∫ –∏—Å—Ç–æ–∫–∞–º $s_i$, –∞ –µ–¥–∏–Ω—Å—Ç–≤–µ–Ω–Ω—ã–π —Å—Ç–æ–∫ $t$ –∞–Ω–∞–ª–æ–≥–∏—á–Ω—ã–º –æ–±—Ä–∞–∑–æ–º –ø–æ—Ç—Ä–µ–±–ª—è–µ—Ç –ø–æ—Ç–æ–∫ –ª—é–±–æ–π –∂–µ–ª–∞–µ–º–æ–π –≤–µ–ª–∏—á–∏–Ω—ã –æ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã—Ö —Å—Ç–æ–∫–æ–≤ $t_i$[@cormen2013].

## –ú–∏–Ω–∏–º–∞–ª—å–Ω—ã–π –æ—Å—Ç–æ–≤

_–û—Å—Ç–æ–≤–Ω—ã–º –¥–µ—Ä–µ–≤–æ–º_ –∏–ª–∏ _–æ—Å—Ç–æ–≤–æ–º_ –≥—Ä–∞—Ñ–∞ $G(V, E)$ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è —Å–≤—è–∑–Ω—ã–π –ø–æ–¥–≥—Ä–∞—Ñ –±–µ–∑ —Ü–∏–∫–ª–æ–≤, —Å–æ–¥–µ—Ä–∂–∞—â–∏–π –≤—Å–µ –≤–µ—Ä—à–∏–Ω—ã –∏—Å—Ö–æ–¥–Ω–æ–≥–æ –≥—Ä–∞—Ñ–∞.
–õ—é–±–æ–µ –æ—Å—Ç–æ–≤–Ω–æ–µ –¥–µ—Ä–µ–≤–æ –≤ –≥—Ä–∞—Ñ–µ —Å $|V| = n$ –≤–µ—Ä—à–∏–Ω–∞–º–∏
—Å–æ–¥–µ—Ä–∂–∏—Ç —Ä–æ–≤–Ω–æ $n-1$ —Ä–µ–±—Ä–æ.

_–ú–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –æ—Å—Ç–æ–≤–Ω–æ–µ –¥–µ—Ä–µ–≤–æ_ ‚Äî —ç—Ç–æ –æ—Å—Ç–æ–≤–Ω–æ–µ –¥–µ—Ä–µ–≤–æ,
—Å—É–º–º–∞ –≤–µ—Å–æ–≤ —Ä–µ–±–µ—Ä –∫–æ—Ç–æ—Ä–æ–≥–æ –º–∏–Ω–∏–º–∞–ª—å–Ω–∞.

_–ó–∞–¥–∞—á–∞ –æ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–º –æ—Å—Ç–æ–≤–µ_: –≤–æ –≤–∑–≤–µ—à–µ–Ω–Ω–æ–º —Å–≤—è–∑–Ω–æ–º –≥—Ä–∞—Ñ–µ –Ω–∞–π—Ç–∏ –æ—Å—Ç–æ–≤ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–≥–æ –≤–µ—Å–∞, —Ç–æ –µ—Å—Ç—å –æ—Å—Ç–æ–≤, —Å—É–º–º–∞—Ä–Ω—ã–π –≤–µ—Å —Ä–µ–±–µ—Ä –∫–æ—Ç–æ—Ä–æ–≥–æ —è–≤–ª—è–µ—Ç—Å—è –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–º[@novikov2008].

## –ê–ª–≥–æ—Ä–∏—Ç–º –ü—Ä–∏–º–∞

–ê–ª–≥–æ—Ä–∏—Ç–º –ü—Ä–∏–º–∞ --- –∞–ª–≥–æ—Ä–∏—Ç–º –ø–æ—Å—Ç—Ä–æ–µ–Ω–∏—è –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–≥–æ –æ—Å—Ç–æ–≤–Ω–æ–≥–æ –¥–µ—Ä–µ–≤–∞ –≤–∑–≤–µ—à–µ–Ω–Ω–æ–≥–æ —Å–≤—è–∑–Ω–æ–≥–æ –Ω–µ–æ—Ä–∏–µ–Ω—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ –≥—Ä–∞—Ñ–∞.

–í –¥–∞–Ω–Ω–æ–º –∞–ª–≥–æ—Ä–∏—Ç–º–µ –∫—Ä–∞—Ç—á–∞–π—à–∏–π –æ—Å—Ç–æ–≤ –ø–æ—Ä–æ–∂–¥–∞–µ—Ç—Å—è –≤ –ø—Ä–æ—Ü–µ—Å—Å–µ —Ä–∞–∑—Ä–∞—Å—Ç–∞–Ω–∏—è –æ–¥–Ω–æ–≥–æ –¥–µ—Ä–µ–≤–∞, –∫ –∫–æ—Ç–æ—Ä–æ–º—É –ø—Ä–∏—Å–æ–µ–¥–∏–Ω—è—é—Ç—Å—è –±–ª–∏–∂–∞–π—à–∏–µ –æ–¥–∏–Ω–æ—á–Ω—ã–µ –≤–µ—Ä—à–∏–Ω—ã.
–ü—Ä–∏ —ç—Ç–æ–º –¥–ª—è –∫–∞–∂–¥–æ–π –≤–µ—Ä—à–∏–Ω—ã $v$, –∫—Ä–æ–º–µ –Ω–∞—á–∞–ª—å–Ω–æ–π,
–∏—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è –¥–≤–µ –ø–æ–º–µ—Ç–∫–∏: $Œ±[v]$ ‚Äî —ç—Ç–æ –±–ª–∏–∂–∞–π—à–∞—è
–∫ $v$ –≤–µ—Ä—à–∏–Ω–∞, —É–∂–µ –≤–∫–ª—é—á—ë–Ω–Ω–∞—è –≤ –æ—Å—Ç–æ–≤, –∞ $Œ≤[v]$ ‚Äî —ç—Ç–æ –¥–ª–∏–Ω–∞ —Ä–µ–±—Ä–∞, —Å–æ–µ–¥–∏–Ω—è—é—â–µ–≥–æ $v$ —Å –æ—Å—Ç–æ–≤–æ–º.
–ï—Å–ª–∏ –≤–µ—Ä—à–∏–Ω—É $v$ –µ—â—ë –Ω–µ–ª—å–∑—è —Å–æ–µ–¥–∏–Ω–∏—Ç—å —Å –æ—Å—Ç–æ–≤–æ–º –æ–¥–Ω–∏–º —Ä–µ–±—Ä–æ–º, —Ç–æ
$Œ±[v] := 0$, $Œ≤[v] := ‚àû$.

–ù–∞ –≤—ã—Ö–æ–¥–µ –º—ã –ø–æ–ª—É—á–∞–µ–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ä–µ–±–µ—Ä, –∫–æ—Ç–æ—Ä—ã–µ –ª–µ–∂–∞—Ç –≤ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–º –æ—Å—Ç–æ–≤–Ω–æ–º –¥–µ—Ä–µ–≤–µ[@novikov2008].

\begin{algorithm}
\caption{–ü—Ä–∏–º–∞, –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏—è –æ—Å—Ç–æ–≤–Ω–æ–≥–æ –¥–µ—Ä–µ–≤–∞ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–π —Å—Ç–æ–∏–º–æ—Å—Ç–∏}\label{alg:prim}
\begin{algorithmic}
\Require –≥—Ä–∞—Ñ $G(V, E)$, –∑–∞–¥–∞–Ω–Ω—ã–π –º–∞—Ç—Ä–∏—Ü–µ–π –¥–ª–∏–Ω —Ä—ë–±–µ—Ä $C$.
\Ensure –º–Ω–æ–∂–µ—Å—Ç–≤–æ $T$ —Ä—ë–±–µ—Ä –∫—Ä–∞—Ç—á–∞–π—à–µ–≥–æ –æ—Å—Ç–æ–≤–∞.
\State \Select $u ‚àà V$ \Comment{–≤—ã–±–∏—Ä–∞–µ–º –ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω—É—é –≤–µ—Ä—à–∏–Ω—É}
\State $S :=\{u\}$ \Comment{$S$ ‚Äî –º–Ω–æ–∂–µ—Å—Ç–≤–æ –≤–µ—Ä—à–∏–Ω, –≤–∫–ª—é—á—ë–Ω–Ω—ã—Ö –≤ –∫—Ä–∞—Ç—á–∞–π—à–∏–π –æ—Å—Ç–æ–≤}
\State $T := ‚àÖ$ \Comment{$T$ ‚Äî –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ä—ë–±–µ—Ä, –≤–∫–ª—é—á—ë–Ω–Ω—ã—Ö –≤ –∫—Ä–∞—Ç—á–∞–π—à–∏–π –æ—Å—Ç–æ–≤}
\For{$v ‚àà V ‚àí u$}
\If{$v ‚àà Œì(u)$}
\State $Œ±[v] := u$ \Comment{$u$ ‚Äî –±–ª–∏–∂–∞–π—à–∞—è –≤–µ—Ä—à–∏–Ω–∞ –æ—Å—Ç–æ–≤–∞}
\State $Œ≤[v] := C[u, v]$ \Comment{$C[u, v]$ ‚Äî –¥–ª–∏–Ω–∞ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–µ–≥–æ —Ä–µ–±—Ä–∞}
\Else
\State $Œ±[v] := 0$ \Comment{–±–ª–∏–∂–∞–π—à–∞—è –≤–µ—Ä—à–∏–Ω–∞ –æ—Å—Ç–æ–≤–∞ –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω–∞}
\State $Œ≤[v] := ‚àû$ \Comment{–∏ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ —Ç–∞–∫–∂–µ –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω–æ}
\EndIf
\EndFor
\For{$i:=1,p‚àí1$}
\State $x := ‚àû$ \Comment{–Ω–∞—á–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –ø–æ–∏—Å–∫–∞ –±–ª–∏–∂–∞–π—à–µ–π –≤–µ—Ä—à–∏–Ω—ã}
\For{$v ‚àà V ‚ßµ S$}
\If{$Œ≤[v] < x$}
\State $w := v$ \Comment{–Ω–∞—à–ª–∏ –±–æ–ª–µ–µ –±–ª–∏–∑–∫—É—é –≤–µ—Ä—à–∏–Ω—É}
\State $x := Œ≤[v]$ \Comment{–∏ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –¥–æ –Ω–µ—ë}
\EndIf
\EndFor
\State $S := S + w$ \Comment{–¥–æ–±–∞–≤–ª—è–µ–º –Ω–∞–π–¥–µ–Ω–Ω—É—é –≤–µ—Ä—à–∏–Ω—É –≤ –æ—Å—Ç–æ–≤}
\State $T := T + (Œ±[w], w)$ \Comment{–¥–æ–±–∞–≤–ª—è–µ–º –Ω–∞–π–¥–µ–Ω–Ω–æ–µ —Ä–µ–±—Ä–æ –≤ –æ—Å—Ç–æ–≤}
\For{$v ‚àà Œì(w)$}
\If{$v ‚àâ S ‚àß Œ≤[v] > C[v, w]$}
\State $Œ±[v] := w$ \Comment{–∏–∑–º–µ–Ω—è–µ–º –±–ª–∏–∂–∞–π—à—É—é –≤–µ—Ä—à–∏–Ω—É –æ—Å—Ç–æ–≤–∞}
\State $Œ≤[v] := C[v, w]$ \Comment{–∏ –¥–ª–∏–Ω—É –≤–µ–¥—É—â–µ–≥–æ –∫ –Ω–µ–π —Ä–µ–±—Ä–∞}
\EndIf
\EndFor
\EndFor
\end{algorithmic}
\end{algorithm}

## –ê–ª–≥–æ—Ä–∏—Ç–º –ö—Ä–∞—Å–∫–∞–ª–∞

–ê–ª–≥–æ—Ä–∏—Ç–º –ö—Ä–∞—Å–∫–∞–ª–∞ --- –∂–∞–¥–Ω—ã–π –∞–ª–≥–æ—Ä–∏—Ç–º –ø–æ—Å—Ç—Ä–æ–µ–Ω–∏—è –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–≥–æ –æ—Å—Ç–æ–≤–Ω–æ–≥–æ –¥–µ—Ä–µ–≤–∞
–≤–∑–≤–µ—à–µ–Ω–Ω–æ–≥–æ —Å–≤—è–∑–Ω–æ–≥–æ –Ω–µ–æ—Ä–∏–µ–Ω—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ –≥—Ä–∞—Ñ–∞.

–ê–ª–≥–æ—Ä–∏—Ç–º \ref{alg:kruskal} —Å–æ—Å—Ç–æ–∏—Ç –∏–∑ –¥–≤—É—Ö —Ñ–∞–∑.

1. –ù–∞ –ø–æ–¥–≥–æ—Ç–æ–≤–∏—Ç–µ–ª—å–Ω–æ–π —Ñ–∞–∑–µ –≤—Å–µ –¥—É–≥–∏ —É–¥–∞–ª—è—é—Ç—Å—è –∏–∑ –¥–µ—Ä–µ–≤–∞ –∏ —É–ø–æ—Ä—è–¥–æ—á–∏–≤–∞—é—Ç—Å—è –ø–æ –≤–æ–∑—Ä–∞—Å—Ç–∞–Ω–∏—é –∏—Ö –≤–µ—Å–æ–≤.
		–í –≥—Ä–∞—Ñ–µ –æ—Å—Ç–∞—é—Ç—Å—è —Ç–æ–ª—å–∫–æ –≤–µ—Ä—à–∏–Ω—ã, –∫–∞–∂–¥–∞—è –∏–∑ –∫–æ—Ç–æ—Ä—ã—Ö –æ–±—Ä–∞–∑—É–µ—Ç –æ—Ç–¥–µ–ª—å–Ω—É—é –∫–æ–º–ø–æ–Ω–µ–Ω—Ç—É —Å–≤—è–∑–Ω–æ—Å—Ç–∏.

2. –í–æ –≤—Ç–æ—Ä–æ–π —Ñ–∞–∑–µ –¥—É–≥–∏ –ø–µ—Ä–µ–±–∏—Ä–∞—é—Ç—Å—è –≤ –ø–æ—Ä—è–¥–∫–µ –≤–æ–∑—Ä–∞—Å—Ç–∞–Ω–∏—è –≤–µ—Å–∞.
		–ï—Å–ª–∏ –Ω–∞—á–∞–ª–æ –∏ –∫–æ–Ω–µ—Ü –æ—á–µ—Ä–µ–¥–Ω–æ–π –¥—É–≥–∏ –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∞—Ç –æ–¥–Ω–æ–π –∏ —Ç–æ–π –∂–µ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–µ —Å–≤—è–∑–Ω–æ—Å—Ç–∏, –¥—É–≥–∞ –∏–≥–Ω–æ—Ä–∏—Ä—É–µ—Ç—Å—è.
		–ï—Å–ª–∏ –∂–µ –æ–Ω–∏ –ª–µ–∂–∞—Ç –≤ —Ä–∞–∑–Ω—ã—Ö –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∞—Ö —Å–≤—è–∑–Ω–æ—Å—Ç–∏. –¥—É–≥–∞ –¥–æ–±–∞–≤–ª—è–µ—Ç—Å—è –∫ –≥—Ä–∞—Ñ—É,
		–∞ —ç—Ç–∏ –¥–≤–µ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç—ã —Å–≤—è–∑–Ω–æ—Å—Ç–∏ –æ–±—ä–µ–¥–∏–Ω—è—é—Ç—Å—è –≤ –æ–¥–Ω—É.
		–ï—Å–ª–∏ —á–∏—Å–ª–æ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç —Å–≤—è–∑–Ω–æ—Å—Ç–∏ –¥–æ–π–¥–µ—Ç –¥–æ 1, —Ü–∏–∫–ª –ø—Ä–µ—Ä—ã–≤–∞–µ—Ç—Å—è.

_–°–ª–æ–∂–Ω–æ—Å—Ç—å –∞–ª–≥–æ—Ä–∏—Ç–º–∞_: $ùí™(q \log p)$, –≥–¥–µ
$p$ --- –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤–µ—Ä—à–∏–Ω,
$q$ --- –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ä–µ–±–µ—Ä –≥—Ä–∞—Ñ–∞.

\begin{algorithm}
\caption{–ö—Ä–∞—Å–∫–∞–ª–∞, –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏—è –æ—Å—Ç–æ–≤–Ω–æ–≥–æ –¥–µ—Ä–µ–≤–∞ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–π —Å—Ç–æ–∏–º–æ—Å—Ç–∏}
\label{alg:kruskal}
\begin{algorithmic}
  \Require
  	—Å–ø–∏—Å–æ–∫ $E$ —Ä—ë–±–µ—Ä –≥—Ä–∞—Ñ–∞ $G$ —Å –¥–ª–∏–Ω–∞–º–∏,
  	—É–ø–æ—Ä—è–¥–æ—á–µ–Ω–Ω—ã–π –≤ –ø–æ—Ä—è–¥–∫–µ –≤–æ–∑—Ä–∞—Å—Ç–∞–Ω–∏—è –¥–ª–∏–Ω.
  \Ensure –º–Ω–æ–∂–µ—Å—Ç–≤–æ $T$ —Ä—ë–±–µ—Ä –∫—Ä–∞—Ç—á–∞–π—à–µ–≥–æ –æ—Å—Ç–æ–≤–∞.
\State $T :=‚àÖ$
\State $k := 1$ \Comment{–Ω–æ–º–µ—Ä —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º–æ–≥–æ —Ä–µ–±—Ä–∞}
\For{$i := 1, p ‚àí 1$}
\While{$z(T + E[k]) > 0$}
\State $k := k + 1$ \Comment{–ø—Ä–æ–ø—É—Å—Ç–∏—Ç—å —ç—Ç–æ —Ä–µ–±—Ä–æ}
\EndWhile
\State $T := T + E[k]$ \Comment{–¥–æ–±–∞–≤–∏—Ç—å —ç—Ç–æ —Ä–µ–±—Ä–æ –≤ SST}
\State $k := k + 1$ \Comment{–∏ –∏—Å–∫–ª—é—á–∏—Ç—å –µ–≥–æ –∏–∑ —Ä–∞—Å—Å–º–æ—Ç—Ä–µ–Ω–∏—è}
\EndFor
\end{algorithmic}
\end{algorithm}
<!-- SST ‚Äî Shortest Spanning Tree ‚Äî —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–æ–µ –æ–±–æ–∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –∫—Ä–∞—Ç—á–∞–π—à–µ–≥–æ –æ—Å—Ç–æ–≤–∞. -->

## –ú–∞—Ç—Ä–∏—á–Ω–∞—è —Ç–µ–æ—Ä–µ–º–∞ –ö–∏—Ä—Ö–≥–æ—Ñ–∞

_–ú–∞—Ç—Ä–∏—Ü–∞ –ö–∏—Ä—Ö–≥–æ—Ñ–∞_ --- –º–∞—Ç—Ä–∏—Ü–∞
$B(G) = (Œ≤_{ij})_{p√óp} = \diag\{\deg v_i\}-A(G)$,
–≥–¥–µ $p$ --- –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤–µ—Ä—à–∏–Ω –≥—Ä–∞—Ñ–∞.
$A(G) = (Œ±_{ij})_{p√óp} = (v_i ‚àà Œì(v_j))$ --- –º–∞—Ç—Ä–∏—Ü–∞ —Å–º–µ–∂–Ω–æ—Å—Ç–∏.
$$ Œ≤_{ij} = \begin{cases}
-1,&  \text{–µ—Å–ª–∏ $v_i$ —Å–º–µ–∂–Ω–∞ —Å $v_j$;}\\
0,  &\text{–µ—Å–ª–∏ $i ‚â† j$, $v_i$ –Ω–µ —Å–º–µ–∂–Ω–∞ —Å $v_j$;}\\
\deg v_i,&  \text{–µ—Å–ª–∏ $i = j$.}
\end{cases}
$$

### –°–≤–æ–π—Å—Ç–≤–∞ –º–∞—Ç—Ä–∏—Ü—ã –ö–∏—Ä—Ö–≥–æ—Ñ–∞

1. –°—É–º–º—ã —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –≤ –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–µ –∏ –∫–∞–∂–¥–æ–º —Å—Ç–æ–ª–±—Ü–µ –º–∞—Ç—Ä–∏—Ü–µ —Ä–∞–≤–Ω—ã 0.
2. –ê–ª–≥–µ–±—Ä–∞–∏—á–µ—Å–∫–∏–µ –¥–æ–ø–æ–ª–Ω–µ–Ω–∏—è –≤—Å–µ—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –º–∞—Ç—Ä–∏—Ü—ã —Ä–∞–≤–Ω—ã –º–µ–∂–¥—É —Å–æ–±–æ–π.
3. –û–ø—Ä–µ–¥–µ–ª–∏—Ç–µ–ª—å –º–∞—Ç—Ä–∏—Ü—ã –ö–∏—Ä—Ö–≥–æ—Ñ–∞ —Ä–∞–≤–µ–Ω –Ω—É–ª—é.

_–ú–∞—Ç—Ä–∏—á–Ω–∞—è —Ç–µ–æ—Ä–µ–º–∞ –ö–∏—Ä—Ö–≥–æ—Ñ–∞_

–ß–∏—Å–ª–æ –æ—Å—Ç–æ–≤–Ω—ã—Ö –¥–µ—Ä–µ–≤—å–µ–≤ –≤ —Å–≤—è–∑–Ω–æ–º –≥—Ä–∞—Ñ–µ $G$ –ø–æ—Ä—è–¥–∫–∞ $p ‚©æ 2$ —Ä–∞–≤–Ω–æ
–∞–ª–≥–µ–±—Ä–∞–∏—á–µ—Å–∫–æ–º—É –¥–æ–ø–æ–ª–Ω–µ–Ω–∏—é $A_{ij}$ –ª—é–±–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ –º–∞—Ç—Ä–∏—Ü—ã –ö–∏—Ä—Ö–≥–æ—Ñ–∞ $B(G)$[@shaporev2006].
$$ A_{ij} = (-1)^{i+j} |M_{ij}|,
$$
–≥–¥–µ $|M_{ij}|$ --- –º–∏–Ω–æ—Ä, –ø–æ–ª—É—á–µ–Ω–Ω—ã–π –∏–∑ –º–∞—Ç—Ä–∏—Ü—ã $B$ –≤—ã—á–µ—Ä–∫–∏–≤–∞–Ω–∏–µ–º $i$-–π —Å—Ç—Ä–æ–∫–∏ –∏ $j$-–≥–æ —Å—Ç–æ–ª–±—Ü–∞.

## –ö–æ–¥ –ü—Ä—é—Ñ–µ—Ä–∞

_–ö–æ–¥ –ü—Ä—é—Ñ–µ—Ä–∞_ --- —ç—Ç–æ —Å–ø–æ—Å–æ–± –≤–∑–∞–∏–º–Ω–æ –æ–¥–Ω–æ–∑–Ω–∞—á–Ω–æ–≥–æ –∫–æ–¥–∏—Ä–æ–≤–∞–Ω–∏—è –ø–æ–º–µ—á–µ–Ω–Ω—ã—Ö –¥–µ—Ä–µ–≤—å–µ–≤ —Å $p$ –≤–µ—Ä—à–∏–Ω–∞–º–∏ —Å –ø–æ–º–æ—â—å—é –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ $p-2$ —Ü–µ–ª—ã—Ö —á–∏—Å–µ–ª –≤ –æ—Ç—Ä–µ–∑–∫–µ $[1,p]$. –¢–æ –µ—Å—Ç—å, –∫–æ–¥ –ü—Ä—é—Ñ–µ—Ä–∞ --- —ç—Ç–æ –±–∏–µ–∫—Ü–∏—è –º–µ–∂–¥—É –≤—Å–µ–º–∏ –æ—Å—Ç–æ–≤–Ω—ã–º–∏ –¥–µ—Ä–µ–≤—å—è–º–∏ –ø–æ–ª–Ω–æ–≥–æ –≥—Ä–∞—Ñ–∞ –∏ —á–∏—Å–ª–æ–≤—ã–º–∏ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—è–º–∏.

\begin{algorithm}
\caption{–ü–æ—Å—Ç—Ä–æ–µ–Ω–∏–µ –∫–æ–¥–∞ –ü—Ä—é—Ñ–µ—Ä–∞ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –¥–µ—Ä–µ–≤–∞}
\label{alg:pruefer-encode}
\begin{algorithmic}
\Require –î–µ—Ä–µ–≤–æ $T(V, E)$ –≤ –ª—é–±–æ–º –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–∏, –≤–µ—Ä—à–∏–Ω—ã –¥–µ—Ä–µ–≤–∞ –ø—Ä–æ–Ω—É–º–µ—Ä–æ–≤–∞–Ω—ã
—á–∏—Å–ª–∞–º–∏ $1..p$ –ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω—ã–º –æ–±—Ä–∞–∑–æ–º.
\Ensure –ú–∞—Å—Å–∏–≤ $A$: \Array{1..$p ‚àí 1$}{1..$p$}
‚Äî –∫–æ–¥ –ü—Ä—é—Ñ–µ—Ä–∞ –¥–µ—Ä–µ–≤–∞ $T$.
\For{$i := 1, p‚àí1$}
\State $v := \min \{k ‚àà V | d(k) = 1\}$ \Comment{–≤—ã–±–∏—Ä–∞–µ–º –≤–∏—Å—è—á—É—é –≤–µ—Ä—à–∏–Ω—É $v$}
\State $A[i] := Œì(v)$ \Comment{–∑–∞–Ω–æ—Å–∏–º –≤ –∫–æ–¥ –Ω–æ–º–µ—Ä –µ–¥–∏–Ω—Å—Ç–≤–µ–Ω–Ω–æ–π –≤–µ—Ä—à–∏–Ω—ã, —Å–º–µ–∂–Ω–æ–π —Å $v$}
\State $V := V ‚àí v$ \Comment{—É–¥–∞–ª—è–µ–º –≤–µ—Ä—à–∏–Ω—É $v$ –∏–∑ –¥–µ—Ä–µ–≤–∞}
\EndFor
\end{algorithmic}
\end{algorithm}

_–°–ª–æ–∂–Ω–æ—Å—Ç—å –∞–ª–≥–æ—Ä–∏—Ç–º–∞_ \ref{alg:pruefer-encode}: $ùí™(p \log p)$, –µ—Å–ª–∏ –ø–æ–∏—Å–∫ –º–∏–Ω–∏–º—É–º–∞ –æ—Å—É—â–µ—Å—Ç–≤–ª—è–µ—Ç—Å—è –∑–∞ $ùí™(\log p)$.

–í–æ—Å—Å—Ç–∞–Ω–∞–≤–ª–µ–Ω–∏–µ –∑–∞–∫–æ–¥–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ –ø–æ –∫–æ–¥—É –ü—Ä—é—Ñ–µ—Ä–∞ –¥–µ—Ä–µ–≤–∞ --- –∞–ª–≥–æ—Ä–∏—Ç–º \ref{alg:pruefer-decode}.

\begin{algorithm}
\caption{–†–∞—Å–ø–∞–∫–æ–≤–∫–∞ –∫–æ–¥–∞ –ü—Ä—é—Ñ–µ—Ä–∞ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –¥–µ—Ä–µ–≤–∞}
\label{alg:pruefer-decode}
\begin{algorithmic}
\Require –ú–∞—Å—Å–∏–≤ $A$: \Array{1..$p ‚àí 1$}{1..$p$}
‚Äî –∫–æ–¥ –ü—Ä—é—Ñ–µ—Ä–∞ –¥–µ—Ä–µ–≤–∞ $T$.
\Ensure –î–µ—Ä–µ–≤–æ $T(V, E)$, –∑–∞–¥–∞–Ω–Ω–æ–µ –º–Ω–æ–∂–µ—Å—Ç–≤–æ–º —Ä—ë–±–µ—Ä $E$, –≤–µ—Ä—à–∏–Ω—ã
–ø—Ä–æ–Ω—É–º–µ—Ä–æ–≤–∞–Ω—ã —á–∏—Å–ª–∞–º–∏ $1..p$.
\State $E : = ‚àÖ$ \Comment{–≤–Ω–∞—á–∞–ª–µ –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ä—ë–±–µ—Ä –ø—É—Å—Ç–æ}
\State $B : = 1..p$ \Comment{–º–Ω–æ–∂–µ—Å—Ç–≤–æ –Ω–µ–∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–Ω—ã—Ö –Ω–æ–º–µ—Ä–æ–≤ –≤–µ—Ä—à–∏–Ω}
\For{$i := 1,p ‚àí 1$}
\State \Comment{–≤—ã–±–∏—Ä–∞–µ–º –≤–µ—Ä—à–∏–Ω—É $v$ ‚Äî –Ω–µ–∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–Ω—É—é –≤–µ—Ä—à–∏–Ω—É —Å –Ω–∞–∏–º–µ–Ω—å—à–∏–º}
\State \Comment{–Ω–æ–º–µ—Ä–æ–º, –∫–æ—Ç–æ—Ä—ã–π –Ω–µ –≤—Å—Ç—Ä–µ—á–∞–µ—Ç—Å—è –≤ –æ—Å—Ç–∞—Ç–∫–µ –∫–æ–¥–∞ –ü—Ä—é—Ñ–µ—Ä–∞}
\State $v : = \min \left\{k ‚àà B \middle| ‚àÄ j ‚©æ i (k ‚â† A[j])\right\}$
\State $E : = E + (v, A[i])$ \Comment{–¥–æ–±–∞–≤–ª—è–µ–º —Ä–µ–±—Ä–æ $(v, A[i])$}
\State $B : = B ‚àí v$ \Comment{—É–¥–∞–ª—è–µ–º –≤–µ—Ä—à–∏–Ω—É $v$ –∏–∑ —Å–ø–∏—Å–∫–∞ –Ω–µ–∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–Ω—ã—Ö}
\EndFor
\end{algorithmic}
\end{algorithm}

_–°–ª–æ–∂–Ω–æ—Å—Ç—å –∞–ª–≥–æ—Ä–∏—Ç–º–∞_ \ref{alg:pruefer-decode}: $ùí™(p \log p)$, –µ—Å–ª–∏ –ø–æ–∏—Å–∫ –º–∏–Ω–∏–º—É–º–∞ –æ—Å—É—â–µ—Å—Ç–≤–ª—è–µ—Ç—Å—è –∑–∞ $ùí™(\log p)$.

## –≠–π–ª–µ—Ä–æ–≤ –≥—Ä–∞—Ñ

–ï—Å–ª–∏ —Å–≤—è–∑–Ω—ã–π –≥—Ä–∞—Ñ –∏–º–µ–µ—Ç —Ü–∏–∫–ª (–Ω–µ –æ–±—è–∑–∞—Ç–µ–ª—å–Ω–æ –ø—Ä–æ—Å—Ç–æ–π),
—Å–æ–¥–µ—Ä–∂–∞—â–∏–π –≤—Å–µ —Ä—ë–±—Ä–∞ –≥—Ä–∞—Ñ–∞ (–∫–∞–∂–¥–æ–µ --- –ø–æ –æ–¥–Ω–æ–º—É —Ä–∞–∑—É),
—Ç–æ —Ç–∞–∫–æ–π —Ü–∏–∫–ª –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è _—ç–π–ª–µ—Ä–æ–≤—ã–º_ —Ü–∏–∫–ª–æ–º,
–∞ –≥—Ä–∞—Ñ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è _—ç–π–ª–µ—Ä–æ–≤—ã–º_ –≥—Ä–∞—Ñ–æ–º.
–≠–π–ª–µ—Ä–æ–≤ —Ü–∏–∫–ª —Å–æ–¥–µ—Ä–∂–∏—Ç –≤—Å–µ –≤–µ—Ä—à–∏–Ω—ã –≥—Ä–∞—Ñ–∞.

_–¢–µ–æ—Ä–µ–º–∞_ –ï—Å–ª–∏ –≥—Ä–∞—Ñ $G$ —Å–≤—è–∑–µ–Ω –∏ –Ω–µ—Ç—Ä–∏–≤–∏–∞–ª–µ–Ω,
—Ç–æ $G$ ‚Äî —ç–π–ª–µ—Ä–æ–≤ –≥—Ä–∞—Ñ $‚ü∫$ –∫–∞–∂–¥–∞—è –≤–µ—Ä—à–∏–Ω–∞ $G$ –∏–º–µ–µ—Ç —á—ë—Ç–Ω—É—é —Å—Ç–µ–ø–µ–Ω—å[@novikov2008].

\begin{algorithm}
\caption{–ø–æ—Å—Ç—Ä–æ–µ–Ω–∏—è —ç–π–ª–µ—Ä–æ–≤–∞ —Ü–∏–∫–ª–∞}
\label{alg:euler_cyc}
\begin{algorithmic}
\Require —ç–π–ª–µ—Ä–æ–≤ –≥—Ä–∞—Ñ $G(V, E)$, –∑–∞–¥–∞–Ω–Ω—ã–π —Å–ø–∏—Å–∫–∞–º–∏ —Å–º–µ–∂–Ω–æ—Å—Ç–∏ ($Œì[v]$ --- —Å–ø–∏—Å–æ–∫ –≤–µ—Ä—à–∏–Ω,
—Å–º–µ–∂–Ω—ã—Ö —Å –≤–µ—Ä—à–∏–Ω–æ–π $v$).
\Ensure –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –≤–µ—Ä—à–∏–Ω —ç–π–ª–µ—Ä–æ–≤–∞ —Ü–∏–∫–ª–∞.
\State $S := ‚àÖ$ \Comment{—Å—Ç–µ–∫ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –≤–µ—Ä—à–∏–Ω}
\State \Select $v ‚àà V$ \Comment{–ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω–∞—è –≤–µ—Ä—à–∏–Ω–∞}
\State $v ‚Üí S$ \Comment{–ø–æ–ª–æ–∂–∏—Ç—å $v$ –≤ —Å—Ç–µ–∫ $S$}
\While{$S ‚â† ‚àÖ$}
\State $v := \Top S$ \Comment{$v$ ‚Äî –≤–µ—Ä—Ö–Ω–∏–π —ç–ª–µ–º–µ–Ω—Ç —Å—Ç–µ–∫–∞}
\If{$Œì[v] = ‚àÖ$}
\State $v ‚Üê S$; \Yield $v$ \Comment{–æ—á–µ—Ä–µ–¥–Ω–∞—è –≤–µ—Ä—à–∏–Ω–∞ —ç–π–ª–µ—Ä–æ–≤–∞ —Ü–∏–∫–ª–∞}
\Else
\State \Select $u ‚àà Œì[v]$ \Comment{–≤–∑—è—Ç—å –ø–µ—Ä–≤—É—é –≤–µ—Ä—à–∏–Ω—É –∏–∑ —Å–ø–∏—Å–∫–∞ —Å–º–µ–∂–Ω–æ—Å—Ç–∏}
\State $u ‚Üí S$ \Comment{–ø–æ–ª–æ–∂–∏—Ç—å u –≤ —Å—Ç–µ–∫}
\State $Œì[v] := Œì[v] ‚àí u$; $Œì[u] := Œì[u] ‚àí v$ \Comment{—É–¥–∞–ª–∏—Ç—å —Ä–µ–±—Ä–æ $(v, u)$}
\EndIf
\EndWhile
\end{algorithmic}
\end{algorithm}

_–°–ª–æ–∂–Ω–æ—Å—Ç—å –∞–ª–≥–æ—Ä–∏—Ç–º–∞_ \ref{alg:euler_cyc}: $ùí™(q)$.

## –ì–∞–º–∏–ª—å—Ç–æ–Ω–æ–≤ –≥—Ä–∞—Ñ

–ï—Å–ª–∏ –≥—Ä–∞—Ñ –∏–º–µ–µ—Ç –ø—Ä–æ—Å—Ç–æ–π —Ü–∏–∫–ª, —Å–æ–¥–µ—Ä–∂–∞—â–∏–π –≤—Å–µ –≤–µ—Ä—à–∏–Ω—ã –≥—Ä–∞—Ñ–∞ (–ø–æ –æ–¥–Ω–æ–º—É —Ä–∞–∑—É), —Ç–æ —Ç–∞–∫–æ–π —Ü–∏–∫–ª –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è _–≥–∞–º–∏–ª—å—Ç–æ–Ω–æ–≤—ã–º_ —Ü–∏–∫–ª–æ–º, –∞ –≥—Ä–∞—Ñ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è _–≥–∞–º–∏–ª—å—Ç–æ–Ω–æ–≤—ã–º_ –≥—Ä–∞—Ñ–æ–º.

–ì–∞–º–∏–ª—å—Ç–æ–Ω–æ–≤ —Ü–∏–∫–ª –Ω–µ –æ–±—è–∑–∞—Ç–µ–ª—å–Ω–æ —Å–æ–¥–µ—Ä–∂–∏—Ç –≤—Å–µ —Ä–µ–±—Ä–∞ –≥—Ä–∞—Ñ–∞. –ì–∞–º–∏–ª—å—Ç–æ–Ω–æ–≤—ã–º –º–æ–∂–µ—Ç –±—ã—Ç—å —Ç–æ–ª—å–∫–æ —Å–≤—è–∑–Ω—ã–π –≥—Ä–∞—Ñ.
